    function startWorkout(dayKey) {

      currentDay = dayKey;

      currentExerciseIndex = 0;

      currentSet = 0;

      timerMode = "idle";

      clearTimer();

      const day = workouts[dayKey];

      noWorkoutMessage.style.display = "none";

      exerciseActive.style.display = "block";

      // עדכון טבלת התרגילים

      planTitleEl.textContent = day.name;

      planSubtitleEl.textContent = "רשימת התרגילים המלאה לאימון זה:";

      planTableEl.style.display = "table";

      planBodyEl.innerHTML = "";

        tr.innerHTML = `

          <td>${ex.reps}</td>

          <td>${ex.restSec} שניות</td>

        `;

        planBodyEl.appendChild(tr);

      });

      // תרגיל ראשון

      loadCurrentExercise();

      updateProgress();

    }

    function loadCurrentExercise() {

      const day = workouts[currentDay];

      const ex = day.exercises[currentExerciseIndex];

      currentSet = 0;

      timerMode = "idle";

      clearTimer();

      timerDisplayEl.textContent = "00:00";

      timerLabelEl.textContent = "מוכן לסט הבא";

      exerciseNameEl.textContent = ex.name;

      exerciseMetaEl.textContent = `${day.name} • תרגיל ${currentExerciseIndex + 1} מתוך ${day.exercises.length}`;

      exerciseDescEl.textContent = ex.desc;

      badgeSetsEl.textContent = `סטים: ${ex.sets}`;

      badgeRepsEl.textContent = `חזרות: ${ex.reps}`;

      badgeRestEl.textContent = `מנוחה בין סטים: ${ex.restSec} שניות`;

      btnStartSet.disabled = false;

      btnFinishSet.disabled = true;

      updateProgress();

    }

    function countTotalSets(dayKey) {

      const exs = workouts[dayKey].exercises;

      return exs.reduce((sum, e) => sum + e.sets, 0);

    }

    function updateProgress() {

      if (!currentDay) {

        progressBarEl.style.width = "0%";

        progressTextEl.textContent = "";

        return;

      }

      const day = workouts[currentDay];

      const totalSets = countTotalSets(currentDay);

      let completedSets = 0;

      for (let i = 0; i < day.exercises.length; i++) {

        const ex = day.exercises[i];

        if (i < currentExerciseIndex) {

          completedSets += ex.sets;

        } else if (i === currentExerciseIndex) {

          completedSets += currentSet;

        }

      }

      const percent = Math.round((completedSets / totalSets) * 100);

      progressBarEl.style.width = percent + "%";

      progressTextEl.textContent = `השלמת בערך ${percent}% מהאימון (סט ${currentSet} בתרגיל ${currentExerciseIndex + 1}).`;

    }

    function startSet() {

      if (!currentDay) return;

      const ex = workouts[currentDay].exercises[currentExerciseIndex];

      timerMode = "work";

      timerRemaining = 0; // אפשר להציג זמן עולה, אבל נשאיר 00:00 – אתה עובד בקצב שלך

      timerLabelEl.textContent = "סט בעבודה – בצע את החזרות לפי התחושה שלך.";

      timerDisplayEl.textContent = "▶️";

      btnStartSet.disabled = true;

      btnFinishSet.disabled = false;

    }

    function finishSet() {

      if (!currentDay) return;

      const ex = workouts[currentDay].exercises[currentExerciseIndex];

      currentSet += 1;

      updateProgress();

      if (currentSet >= ex.sets) {

        // כל הסטים לתרגיל זה הושלמו

        goToNextExercise();

      } else {

        // מנוחה בין סטים

        startRest(ex.restSec);

      }

    }

    function startRest(seconds) {

      timerMode = "rest";

      timerRemaining = seconds;

      timerLabelEl.textContent = "מנוחה בין סטים – נצל את הזמן לשתות מים.";

      updateTimerDisplay();

      clearTimer();

      timerInterval = setInterval(() => {

        timerRemaining -= 1;

        if (timerRemaining <= 0) {

          clearTimer();

          timerMode = "idle";

          timerLabelEl.textContent = "המנוחה הסתיימה – מוכן לסט הבא.";

          timerDisplayEl.textContent = "00:00";

          btnStartSet.disabled = false;

          btnFinishSet.disabled = true;

        } else {

          updateTimerDisplay();

        }

      }, 1000);

      btnStartSet.disabled = true;

      btnFinishSet.disabled = true;

    }

    function updateTimerDisplay() {

      const min = String(Math.floor(timerRemaining / 60)).padStart(2, "0");

      const sec = String(timerRemaining % 60).padStart(2, "0");

      timerDisplayEl.textContent = `${min}:${sec}`;

    }

    function clearTimer() {

      if (timerInterval) {

        clearInterval(timerInterval);

        timerInterval = null;

      }

    }

    function goToNextExercise() {

      const day = workouts[currentDay];

      currentExerciseIndex += 1;

      if (currentExerciseIndex >= day.exercises.length) {

        // אימון הסתיים

        clearTimer();

        timerMode = "idle";

        timerDisplayEl.textContent = "✔️";

        timerLabelEl.textContent = "כל הכבוד! סיימת את כל האימון להיום.";

        btnStartSet.disabled = true;

        btnFinishSet.disabled = true;

        btnSkipExercise.disabled = true;

        updateProgress();

      } else {

        loadCurrentExercise();

      }

    }

    function skipExercise() {

      if (!currentDay) return;

      const day = workouts[currentDay];

      currentExerciseIndex += 1;

      if (currentExerciseIndex >= day.exercises.length) {

        // אין עוד תרגילים

        goToNextExercise();

      } else {

        loadCurrentExercise();

      }

    }

    function resetWorkout() {

      clearTimer();

      currentDay = null;

      currentExerciseIndex = 0;

      currentSet = 0;

      timerMode = "idle";

      timerDisplayEl.textContent = "00:00";

      timerLabelEl.textContent = "מוכן לסט הבא";

      exerciseActive.style.display = "none";

      noWorkoutMessage.style.display = "block";

      planTitleEl.textContent = "תוכנית היום";

      planSubtitleEl.textContent = "בחר יום אימון כדי לראות את רשימת התרגילים המלאה.";

      planTableEl.style.display = "none";

      planBodyEl.innerHTML = "";

      progressBarEl.style.width = "0%";

      progressTextEl.textContent = "";

      btnSkipExercise.disabled = false;

    }

        Object.keys(tabContents).forEach(key => {

          tabContents[key].style.display = key === tab ? "block" : "none";

        });

      });

    });
